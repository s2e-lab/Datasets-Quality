def sort_numbers(numbers: str) -> str:
    numbers_mapping = {
        'zero': 0,
        'one': 1,
        'two': 2,
        'three': 3,
        'four': 4,
        'five': 5,
        'six': 6,
        'seven': 7,
        'eight': 8,
        'nine': 9
    }
    numbers_list = numbers.split()
    numbers_values = [numbers_mapping[number] for number in numbers_list]
    sorted_numbers_values = sorted(numbers_values)
    sorted_numbers = [number for number, value in numbers_mapping.items() if value in sorted_numbers_values]
    return ' '.join(sorted_numbers)
